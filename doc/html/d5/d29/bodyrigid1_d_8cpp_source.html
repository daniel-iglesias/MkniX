31 RigidBar::RigidBar( std::string title_in,
32  Node * nodeA_in,
33  Node * nodeB_in,
34  double mass_in
35  )
36  : RigidBody( title_in )
37 {
38  this->mass = mass_in;
39  // Read nodes from theReader:
40  // It is assumed that correspond to extremes of bar and
41  // it is necessary to change them to CoG and director vector
42  nodes.push_back( nodeA_in );
43  nodes.push_back( nodeB_in );
44  // Create the boundary nodes and assign them the support nodes and constant function
45  domainNodes.push_back( new Node(nodeA_in) );
46  domainNodes.push_back( new Node(nodeB_in) );
47  // We calculate bar properties and reposition nodes to the normalized position
48  lenght = std::sqrt( std::pow(nodeB_in->getx() - nodeA_in->getx(), 2) +
49  std::pow(nodeB_in->gety() - nodeA_in->gety(), 2) +
50  std::pow(nodeB_in->getz() - nodeA_in->getz(), 2) );
51  rho = mass_in / lenght ;
52  // Center of mass is node 0
53  double x_0 = 0.5*( nodeB_in->getx() + nodeA_in->getx() );
54  double y_0 = 0.5*( nodeB_in->gety() + nodeA_in->gety() );
55  double z_0 = 0.5*( nodeB_in->getz() + nodeA_in->getz() );
56  // First director vector is node 1
57  double x_1 = x_0 + ( nodeB_in->getx() - nodeA_in->getx() ) / lenght;
58  double y_1 = y_0 + ( nodeB_in->gety() - nodeA_in->gety() ) / lenght;
59  double z_1 = z_0 + ( nodeB_in->getz() - nodeA_in->getz() ) / lenght;
60  // Now we reset the positions of the nodes (stored in the general configuration)
61  nodes[0]->setX(x_0); nodes[0]->setY(y_0); nodes[0]->setZ(z_0);
62  nodes[1]->setX(x_1); nodes[1]->setY(y_1); nodes[1]->setZ(z_0);
63 
64  // Time to define the relationship between boundary nodes and formulation nodes
65  double end_int = this->domainNodes.size();
66 #pragma omp parallel for
67  for (int i=0;
68  i < end_int;
69  ++i)
70  {
71  this->domainNodes[i]->addSupportNode( this->nodes[0] );
72  this->domainNodes[i]->addSupportNode( this->nodes[1] );
73 
74  this->domainNodes[i]->setShapeFunType("BAR");
75  this->domainNodes[i]->shapeFunSolve("BAR", 1.);
76  }
77 
78  this->localMassMatrix.resize( 2*dim, 2*dim );
79  this->externalForces.resize( 2*dim );
80 
81 }
