32 RigidBar::RigidBar(std::string title_in,
33  Node * nodeA_in,
34  Node * nodeB_in,
35  double mass_in
36 )
37  : RigidBody(title_in)
38 {
39  this->mass = mass_in;
40  // Read frameNodes from theReader:
41  // It is assumed that correspond to extremes of bar and
42  frameNodes.push_back(nodeA_in);
43  frameNodes.push_back(nodeB_in);
44  // We calculate bar properties and reposition frameNodes to the normalized position
45  lenght = std::sqrt(std::pow(nodeB_in->getX() - nodeA_in->getX(), 2) +
46  std::pow(nodeB_in->getY() - nodeA_in->getY(), 2) +
47  std::pow(nodeB_in->getZ() - nodeA_in->getZ(), 2));
48  rho = mass_in / lenght;
49

75 void RigidBar::calcMassMatrix()
76 {
77  this->localMassMatrix(0, 0) = 1. / 3. * mass;
78  this->localMassMatrix(1, 1) = 1. / 3. * mass;
79  this->localMassMatrix(Simulation::getDim(), Simulation::getDim()) = 1. / 3. * mass;
80  this->localMassMatrix(Simulation::getDim() + 1, Simulation::getDim() + 1) = 1. / 3. * mass;
81 
82  this->localMassMatrix(0, Simulation::getDim()) = 1. / 6. * mass;
83  this->localMassMatrix(Simulation::getDim(), 0) = 1. / 6. * mass;
84  this->localMassMatrix(1, Simulation::getDim() + 1) = 1. / 6. * mass;
85  this->localMassMatrix(Simulation::getDim() + 1, 1) = 1. / 6. * mass;
86 
87  if (Simulation::getDim() == 3) {
88  this->localMassMatrix(2, 2) = 1. / 3. * mass;
89  this->localMassMatrix(Simulation::getDim() + 2, Simulation::getDim() + 2) = 1. / 3. * mass;
90  this->localMassMatrix(2, Simulation::getDim() + 2) = 1. / 6. * mass;
91  this->localMassMatrix(Simulation::getDim() + 2, 2) = 1. / 6. * mass;
