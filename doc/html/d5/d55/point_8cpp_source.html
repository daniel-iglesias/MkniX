72 {
73  double di = alphai * dc;

84  it != domainNodes.end();

86  // enclose first in a rectangle, seems more efficient:
87  if ( (*it)->getx() >= minX && (*it)->getx() <= maxX ){
88  if ( (*it)->gety() >= minY && (*it)->gety() <= maxY ){
89  // distance criteria:
90  distance = sqrt( pow((*it)->getx()-X,2)
91  +pow((*it)->gety()-Y,2)
92  +pow((*it)->getz()-Z,2) );
93  if ( distance <= di ){
94 // cout << "Distance = " << distance << endl;
95  supportNodes.push_back( (*it) );
96  if (distance > smax) smax = distance;
97  }
98  }
99  }
100  }
101 // cout << "Old alphai = " << alphai;
102  alphai = smax / dc;
103 // cout << ", New alphai = " << alphai << endl;
