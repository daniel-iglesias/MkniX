97 void RigidBody::outputStep
98 ( const lmx::Vector< data_type > & q, const lmx::Vector< data_type > & qdot )
99 {
100  int nodesSize = domainNodes.size();
101  int i, n, j, m;
102  boundaryConf.push_back( new lmx::Vector<data_type>( dim*nodesSize ) );
103  for( i=0; i<nodesSize; ++i ){
104  for (j=0; j<dim; ++j ){
105  boundaryConf.back()->writeElement( domainNodes[i]->getConf(j), dim*i+j );
106  }
107  }
108 
109  if( computeEnergy )
110  {
111  /*potential, kinetic, elastic, total*/
112  energy.push_back( new lmx::Vector<data_type>( 4 ) );
113  energy.back()->fillIdentity( 0. );
114 
115  nodesSize = nodes.size();
116 
117  for( i=0; i<nodesSize; ++i )
118  {
119  for( n=0; n<Simulation::getDim() ; ++n)
120  {
121  /*potential*/
122  energy.back()->operator()(0)
123  -= q.readElement( Simulation::getDim()*nodes[i]->getNumber() + n )
124  * externalForces.readElement(Simulation::getDim()*i + n);
125 
126  /*kinetic*/
127  for (j=0; j<nodesSize; ++j)
128  {
129  for (m=0; m<Simulation::getDim(); ++m)
130  {
131  energy.back()->operator()(1)
132  += 0.5
133  * qdot.readElement( Simulation::getDim()*nodes[i]->getNumber() + n )
134  * localMassMatrix.readElement( Simulation::getDim()*i + n, Simulation::getDim()*j + m)
135  * qdot.readElement( Simulation::getDim()*nodes[j]->getNumber() + m );
136  }
137  }
138  }
139  }
140  //no Elastic
141  /*Total*/
142  energy.back()->operator()(3) = energy.back()->readElement(0)
143  + energy.back()->readElement(1)
144  + energy.back()->readElement(2);
145  }
