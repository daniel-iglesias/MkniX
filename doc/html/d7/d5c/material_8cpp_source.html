60 void Material::computeD()
61 {
62  double comFacD; // Common factor for matrix D.
63 
64  if (dim == 2) {
65  D.resize(3, 3);
66  // Plain stress case:
67  comFacD = young / (1 - std::pow(poisson, 2)); // = E/(1-mu^2)
68  D(0, 0) = 1;
69  D(0, 1) = poisson;
70  D.writeElement(D.readElement(0, 1), 1, 0);
71  D(1, 1) = 1;
72  D(2, 2) = (1 - poisson) / 2;
73  D *= comFacD; //Apply common factor.
74  }
75  else if (dim == 3) {
76  D.resize(6, 6);
77  comFacD = young / ((1 + poisson) * (1 - 2 * poisson)); // = E/((1+mu)(1-2*mu))
78  D.writeElement(1. - poisson, 0, 0);
79  D.writeElement(1. - poisson, 1, 1);
80  D.writeElement(1. - poisson, 2, 2);
81  D.writeElement(poisson, 0, 1);
82  D.writeElement(poisson, 0, 2);
83  D.writeElement(poisson, 1, 2);
84  D.writeElement(D.readElement(0, 1), 1, 0);
85  D.writeElement(D.readElement(0, 2), 2, 0);
86  D.writeElement(D.readElement(2, 1), 1, 2);
87  D.writeElement(0.5 - poisson, 3, 3);
88  D.writeElement(0.5 - poisson, 4, 4);
89  D.writeElement(0.5 - poisson, 5, 5);
90  D *= comFacD; //Apply common factor.
91  }
92 }

139 void Material::computeS(cofe::TensorRank2Sym<2, double>& S,
140  const cofe::TensorRank2<2, double>& F,
141  double temperature_in)
142 {
143  cofe::TensorRank2Sym<2, double> one;
144  one.beUnityTensor();
145  // E = 1/2 (F'F - 1)
146  E.beTraProductOf(F, F);
147  E -= one;
148  E *= 0.5;
149  // Adding thermal expansion
150  one *= beta * temperature_in;
151  E -= one;
152  one.beUnityTensor();
153  // St. Venant Kirchoff:
154  // S = lambda*tr(E)*1 + 2*mu*E
155  S = E;
156  S *= 2 * mu;
157  one *= lambda * E.trace();
158  S += one;

183 double Material::computeEnergy(const cofe::TensorRank2<2, double>& F)
184 {
185  cofe::TensorRank2Sym<2, double> one;
186  one.beUnityTensor();
187  // E = 1/2 (F'F - 1)
188  E.beTraProductOf(F, F);
189  E -= one;
190  E *= 0.5;
191 
192  // W = 1/2 lambda * tr(E)^2 + mu * E:E
193  double energy = 0.5 * lambda * pow(E.trace(), 2) + mu * E.dot(E);
194  return energy;
195 }
