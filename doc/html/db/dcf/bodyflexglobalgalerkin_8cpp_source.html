161 void FlexGlobalGalerkin::outputStep(const lmx::Vector<data_type>& q, const lmx::Vector<data_type>& qdot)
162 {
163  Body::outputStep();
164  int stressVectorSize = (Simulation::getDim() == 2) ? 3 : 6;
165 
166  if (computeStress) {
167  if (formulation == "LINEAR") {
168 
169  stresses.push_back(lmx::Vector<data_type>(stressVectorSize * nodes.size()));
170 
171  auto end_int = this->cells.size();
172  for (auto i = 0u; i < end_int; ++i) {
173  this->cells[i]->assembleRGaussPoints(stresses.back(), nodes[0]->getNumber());
174  }
175  }
176  else if (formulation == "NONLINEAR") {
177 
178  stresses.push_back(lmx::Vector<data_type>(stressVectorSize * nodes.size()));
179 
180  auto end_int = this->cells.size();
181  for (auto i = 0u; i < end_int; ++i) {
182  this->cells[i]->assembleNLRGaussPoints(stresses.back(), nodes[0]->getNumber());
183  }
184  }
185  recoverStressField(stressVectorSize);
186  }
187 
188  if (computeEnergy) {
189  energies.push_back(new lmx::Vector<data_type>(4)); //potential, kinetic, elastic, total
190 
191  energies.back()->fillIdentity(0.);
192 
193  auto end_int = this->cells.size();
194 
195  for (auto i = 0u; i < end_int; ++i) {
196  energies.back()->operator()(0) += this->cells[i]->calcPotentialEGaussPoints(q); //potential
197  energies.back()->operator()(1) += this->cells[i]->calcKineticEGaussPoints(qdot); //kinetic
198  energies.back()->operator()(2) += this->cells[i]->calcElasticEGaussPoints(); //elastic
199 //total
200  }
201  energies.back()->operator()(3) +=
202  energies.back()->readElement(0) + energies.back()->readElement(1) + energies.back()->readElement(2);
203  }
204 }
