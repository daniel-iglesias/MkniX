157  if(dim==2)

159  else if(dim==3)
160  rk.resize(4, nn);
161  double rr, rr2, xn, yn, zn;
162 
163  // For each derivative, first the rhs is calculated (= d^n/dx^n{ R(x) Pm(x) }^T) and then the linear system is solved to get {phi(n)}
164  if (dim != 2)
165  std::cout << "Warning, space dimension is partially implemented."
166  << std::endl;
167  // Switch type of RBF:
168  switch (rbfType) {
169  case 0:
170  // rbfType = 0 -> MQ
171  // Compute R(i,j) = ((x_i - x_j)+(y_i - y_j) + (alpha_c*d_c)^2 )^q
172  for (int j=0; j<nn; ++j) {
173  xn = gp->supportNodes[j]->getX();
174  yn = gp->supportNodes[j]->getY();
175  zn=0;
176 // zn = gp->supportNodes[j]->getZ();
177  rr2 = std::pow(xp-xn, 2) + std::pow(yp-yn, 2)+ std::pow(zp-zn, 2);
178  rk(0,j) = std::pow( rr2 + std::pow(alpha_c * d_c, 2 ), q);
179  rk(1,j) = 2. * q * std::pow( rr2 + std::pow(alpha_c * d_c, 2) , q-1. ) * ( xp - xn );
180  rk(2,j) = 2. * q * std::pow( rr2 + std::pow(alpha_c * d_c, 2) , q-1. ) * ( yp - yn );
181  if(dim==2) {
182  rk(3,j) = 2. * q * std::pow( rr2 + std::pow(alpha_c * d_c, 2) , q-1 ) +
183  4 * q * (q-1) * std::pow( rr2 + std::pow(alpha_c * d_c, 2) , q-2 ) * std::pow( xp - xn, 2 );
184  // TODO: Check sign of derivatives
185  rk(4,j) = -4 * q * (q-1) * ( xp - xn ) * ( yp - yn ) *
186  std::pow( rr2 + std::pow(alpha_c * d_c, 2) , q-2 );
187  rk(5,j) = 2 * q * std::pow( rr2 + std::pow(alpha_c * d_c, 2) , q-1 ) +
188  4 * q * (q-1) * std::pow( rr2 + std::pow(alpha_c * d_c, 2) , q-2 ) * std::pow( yp - yn, 2 );
189  }
190  else if(dim==3)
191  rk(3,j) = 2. * q * std::pow( rr2 + std::pow(alpha_c * d_c, 2) , q-1. ) * ( zp - zn );
192  }
193  break;
194 
195  case 1:
196  {
197  // rbfType = 1 -> EXP
198  // Compute R(i,j) = exp( -alpha_c*( (x_i - x_j)^2+(y_i - y_j)^2 ) / d_c^2 )
199 // double qq = alpha_c / std::pow(d_c, 2);
200  double qq = 1. / std::pow(alpha_c*d_c*0.4, 2);
201  for (int j=0; j<nn; ++j) {

203  yn = gp->supportNodes[j]->getY();
204  zn = gp->supportNodes[j]->getZ();

206  rk(0,j) = std::exp( -qq * rr2);
207  rk(1,j) = 2 * qq * std::exp( -qq * rr2) * ( xp - xn );
208  rk(2,j) = 2 * qq * std::exp( -qq * rr2) * ( yp - yn );
209  if(dim==2) {

213  rk(5,j) = 2 * qq * std::exp( -qq * rr2)
214  + 4 * qq * qq * std::pow( yp - yn, 2 ) * std::exp( -qq * rr2);
215  }
216  else if(dim==3) {
