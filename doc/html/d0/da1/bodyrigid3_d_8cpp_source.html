88 void RigidBody3D::calcMassMatrix()
89 { // TODO: Define matrix in generic inertia axes
90  if( densityFactor != 1) {
91  mass *= densityFactor;
92  Ixx *= densityFactor;
93  Iyy *= densityFactor;
94  Izz *= densityFactor;
95  Ixy *= densityFactor;
96  Iyz *= densityFactor;
97  Ixz *= densityFactor;
98  }
99  if( Ixy == 0 && Iyz == 0 && Ixz == 0 ) { // Principal inertia axes defined
100  Pxx = .5*( -Ixx +Iyy +Izz );
101  Pyy = .5*( +Ixx -Iyy +Izz );
102  Pzz = .5*( +Ixx +Iyy -Izz );
103 
104  /* For Problem in 3D
105  * MassMatrix = | m00*I3 m01*I3 m02*I3 m03*I3 |
106  * | m10*I3 Pxx*I3 0*I3 0*I3 |
107  * | m20*I3 0*I3 Pyy*I3 0*I3 |
108  * | m30*I3 0*I3 0*I3 Pzz*I3 |
109  */
110 
111  /* m00*I3 */
112  this->localMassMatrix( 0, 0) = mass +Pxx +Pyy +Pzz ;
113  this->localMassMatrix( 1, 1) = mass +Pxx +Pyy +Pzz ;
114  this->localMassMatrix( 2, 2) = mass +Pxx +Pyy +Pzz ;
115 
116  /* m01*I3 */
117  this->localMassMatrix( 0, 3) = -Pxx ;
118  this->localMassMatrix( 1, 4) = -Pxx ;
119  this->localMassMatrix( 2, 5) = -Pxx ;
120 
121  /* m02*I3 */
122  this->localMassMatrix( 0, 6) = -Pyy ;
123  this->localMassMatrix( 1, 7) = -Pyy ;
124  this->localMassMatrix( 2, 8) = -Pyy ;
125 
126  /* m03*I3 */
127  this->localMassMatrix( 0, 9) = -Pzz ;
128  this->localMassMatrix( 1,10) = -Pzz ;
129  this->localMassMatrix( 2,11) = -Pzz ;
130 
131  /* m10*I3 */
132  this->localMassMatrix( 3, 0) = -Pxx ;
133  this->localMassMatrix( 4, 1) = -Pxx ;
134  this->localMassMatrix( 5, 2) = -Pxx ;
135 
136  /* m20*I3 */
137  this->localMassMatrix( 6, 0) = -Pyy ;
138  this->localMassMatrix( 7, 1) = -Pyy ;
139  this->localMassMatrix( 8, 2) = -Pyy ;
140 
141  /* m30*I3 */
142  this->localMassMatrix( 0, 0) = -Pzz ;
143  this->localMassMatrix(10, 1) = -Pzz ;
144  this->localMassMatrix(11, 2) = -Pzz ;
145 
146  /* Pxx*I3 */
147  this->localMassMatrix( 3, 3) = Pxx ;
148  this->localMassMatrix( 4, 4) = Pxx ;
149  this->localMassMatrix( 5, 5) = Pxx ;
150 
151  /* Pyy*I3 */
152  this->localMassMatrix( 6, 6) = Pyy ;

155
