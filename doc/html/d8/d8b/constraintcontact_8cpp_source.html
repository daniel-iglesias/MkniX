55 void ConstraintContact::calcPhi()
56 {
57  normal[0] = +( nodes[1]->getConf(1) - nodes[0]->getConf(1) );
58  normal[1] = -( nodes[1]->getConf(0) - nodes[0]->getConf(0) );
59  double normal_len = std::sqrt( normal[0]*normal[0] + normal[1]*normal[1] );
60  normal[0] /= normal_len;
61  normal[1] /= normal_len;
62  //gap = rt = ||(P-Q1) · normal|| = ||(P-Q2) · normal||
63  // = ||0.5*( ((P-Q1) + (P-Q2)) · normal)||
64  rh = (nodes[2]->getConf(0) - nodes[1]->getConf(0) ) * normal[0]
65  +(nodes[2]->getConf(1) - nodes[1]->getConf(1) ) * normal[1];
66  rt = (nodes[2]->getConf(0) - nodes[0]->getConf(0) ) * normal[0]
67  + (nodes[2]->getConf(1) - nodes[0]->getConf(1) ) * normal[1];
68 
69  this->phi[0] = rt*rt ;
70 // if( rt < 0. ) cout << endl << "NEGATIVE GAP: " << rt << ", " << rh << endl;
71 }
