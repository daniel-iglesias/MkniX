191 void ShapeFunctionMLS::computeMomentMatrix()
192 {
193  // Example with mm=1 (first order):
194  // [A] = P^T W P
195  // Dimensions in 2D...
196  // [A]_{3x3} = [P^T]_{3 x nn} W_{nn x nn} P_{nn x 3}
197  // index mult:
198  // A_{ij} = w_k*p_i(x_k)*p_j(x_k)
199 
200  if(mm == 1) {
201  // A = sum_i w(i)*ppt(i)
202  // with:
203  // ppt(i) = p*p^T = [[1 xi yi] [xi xi^2 xi*yi] [yi xi*yi yi^2]]
204  for (int i=0; i<m; ++i) {
205  for (int j=0; j<m; ++j) {
206  for (int k=0; k<nn; ++k) {
207  A[0].addElement( w.readElement(0,k) *
208  P.readElement(k,i) *
209  P.readElement(k,j)
210  , i, j
211  );
212 // cout << w.readElement(0,k) << " * "
213 // << P.readElement(i,k) << " * "
214 // << P.readElement(j,k) << " = "
215 // << A[0](i,j) << endl;
216  }
217  }
218  }
219  //DERIVATIVES FOR X and Y... only for mm=1 (vamos, chapucilla...)
220  // A,x = sum_i ( w(i),x * ppt(i) )
221  for (int i=0; i<m; ++i) {
222  for (int j=0; j<m; ++j) {
223  for (int k=0; k<nn; ++k) {
224  A[1].addElement( w.readElement(1,k) * // derivada x de weight(k)
225  P.readElement(k,i) *
226  P.readElement(k,j)
227  , i, j
228  );
229  }
230  }
231  }
232  if( dim >= 2 ) {
233  // A,y = sum_i ( w(i),y * ppt(i) )
234  for (int i=0; i<m; ++i) {
235  for (int j=0; j<m; ++j) {
236  for (int k=0; k<nn; ++k) {
237  A[2].addElement( w.readElement(2,k) * // derivada y de weight(k)
238  P.readElement(k,i) *
239  P.readElement(k,j)
240  , i, j
241  );
242  }
243  }
244  }
245  }
246  if( dim == 3 ) {
247  // A,z = sum_i ( w(i),z * ppt(i) )
248  for (int i=0; i<m; ++i) {
249  for (int j=0; j<m; ++j) {
250  for (int k=0; k<nn; ++k) {
251  A[3].addElement( w.readElement(3,k) * // derivada z de weight(k)
252  P.readElement(k,i) *
253  P.readElement(k,j)
254  , i, j
255  );
256  }
257  }
258  }
259  }
260  // Now we compute the RHS: B_I
261  // => B_I(i) = P^T * W = w_I * P_Ii
262  // And the derivatives of the RHS: B_I,x
263  // => B_I(i) = P^T * W,x = w_I,x * P_Ii
264  for (int j=0; j<nn; ++j) {
265  for (int i=0; i<m; ++i) {
266  for (int r=0; r<m; ++r) {
267  B[r][j].writeElement( w.readElement(r,j) * P.readElement(j,i), i );
268  }
269  }
270  }
271  }
272  else std::cout << "WARNING: order not implemented." << std::endl;

299  lsys.solveYourself();
